Math Operators

							x - y   # Difference of x and y

							x * y 	# Product of x and y

							x % y 	# Remainder of x / y
							x ** y	# x to the y power

							# Finding the length of an Array

							# Adding a car/cars
							cars.append("Toyota")    # appends a Toyota to end

							# removing a car
							cars.remove[2]
							del cars[2]

Conditionals


	b = True
	if b == True:
	  print('b is True')

	# Can also be expressed
	if b:
	  print('b is True')

Else


							# Use else to ensure a response
							b = True
							if b:
							    print('b is True')
							else:
							    print('b is False')

More than Booleans


							# Use Comparison Operators
							num = 7
							if num > 5:
							  print('num is greater than 5')

							# double equals ’==’ operator to check if one thing equals another.
							t = 'this is text'
							if t == 'this is text':
							  print('the text matches')

Comparison Operators


								== # equal
								!= # not equal
								>  # greater than
								>= # greater than or equal
								<  # less than
								<= # less than or equal

Else-If (Elif)


							num1 = 3
							num2 = 7

							if num1 > 5:
							  print('num1 is greater than 5')
							elif num2 > 5:
							  print('num2 is greater than 5')
							else:
							  print("they're both too small!")

Combining Conditionals


							num1 = 3
							num2 = 7

							if num1 < 5 and num2 < 5:
							  print("they're both too small!")

							if num1 < 5 or num2 < 5:
							  print("at least one of them is too small!")

Loops (the begininng of automation)


							# Create an Array
							fruits = ['apples', 'oranges', 'bananas']

							# Iterate over the array
							for x in fruits:
							  print(x)

							# You can make it prettier
							for fruit in fruits:
							  print(fruit)

Range


							# Use a range to control a loop
							for i in range(5):
								print('Hello')

How clever can you get?


							# Array of Names
							names = ["John", "Sarah", "Adam", "Crystal", "Jake", "Jim", "Bob"]

							for i in range(7):
								print('Hello', names[i])

Even more clever


							# What if the array changes?
							names = [John, Sarah, Adam, Crystal, Jake, Jim, Bob, Jennifer, Alex]

							# This loop is self adjusting ;)
							for i in range(len(names)):
								print('Hello', names[i])
